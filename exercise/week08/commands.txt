#DEPLOYMENT — VERCEL (frontend)

cd "/Users/administrador/Documents/Coda/cahier-de-vacances-b3/exercise/week08/frontend"

#Install the Vercel CLI globally on the system
npm install -g vercel

#Log in to the Vercel account from the terminal
vercel login

#Link this folder to a Vercel project (creates a new one if none exists; interactive wizard)
vercel

#Add an environment variable (VITE_API_URL) to the project's production environment
vercel env add VITE_API_URL production

#List the project's configured environment variables, to verify it was saved correctly
vercel env ls

#Download the production environment variables to a local file, to inspect them directly
vercel env pull .env.production.local --environment=production

#Deploy the project to production (first attempt)
vercel --prod

#Deploy to production forcing a clean build, ignoring any previous cache
vercel --prod --force
